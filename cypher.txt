working code to for Bangladesh

LOAD CSV WITH HEADERS FROM 'https://github.com/towableslow/kgraph/blob/d0cf266c9b22977370ec3e58291dec24a771eb4e/bangladesh.csv?raw=1' AS csvLine

MERGE (o:Port_of_origin {name: csvLine.Origin, Country: csvLine.Country})
MERGE (c1:Chokepoint {Chokepoint:csvLine.Chokepoint1})
MERGE (c4:Chokepoint {Chokepoint:csvLine.Chokepoint4})
MERGE (c5:Chokepoint {Chokepoint:csvLine.Chokepoint5})

MERGE (s1:Stop {Stop:csvLine.Stop1})
MERGE (s2:Stop {Stop:csvLine.Stop2})

MERGE (d:Destination {Destination:csvLine.Destination})

MERGE (o)-[:From_Chittagong]->(c1)
MERGE (c1)-[:From_Chittagong]->(s1)
MERGE (s1)-[:From_Chittagong]->(s2)
MERGE (s2)-[:From_Chittagong]->(c4)
MERGE (c4)-[:From_Chittagong]->(c5)
MERGE (c5)-[:From_Chittagong]->(d)

RETURN *
